# AnalysisTool

Installation:

    cmsrel CMSSW_7_2_5
    cd CMSSW_7_2_5/src/
    cmsenv

    git clone https://github.com/ekenn003/AnalysisTool.git
    # git clone git@github.com:ekenn003/AnalysisTool.git

    cd $CMSSW_BASE/src/AnalysisTool/AnalysisTool

    ./configure --prefix=($pwd)
    make
    make install
